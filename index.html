    //comentários no c# sao realizados de 2 maneiras com as duas barras para a direita
    public class Pessoa
    {
        private string _nome="";
        //Construtor da class Pessoa
        public Pessoa()
        {}
        //Construtor da class Pessoa com o nome da pessoa
        public Pessoa(string nome)
        {
            _nome = nome;
        }
                        

    /*
    ou então quando queremos comentar varias linhas podemos usar assim  
    Atalhos : CTRL+K+C  e para descomentar CTRL+K+U
    Estes atalhos podem ser configurados em Tools>Options>Environment>Keyboard
    */
        public string Nome
        {
            get { return _nome; }
            set{ _nome=value;}
        }
    }
